    <script>

        // 定義 60 年循環的基礎數據 (根據天干地支規律推算)

        // 1924 為 甲子 (0)

        // 1927 為 丁卯 (3)

        

        const stems = ["甲", "乙", "丙", "丁", "戊", "己", "庚", "辛", "壬", "癸"];

        const branches = ["子", "丑", "寅", "卯", "辰", "巳", "午", "未", "申", "酉", "戌", "亥"];

        

        // 司天對應表 (地支)

        const siTianMap = {

            "子": "少陰君火", "午": "少陰君火",

            "丑": "太陰濕土", "未": "太陰濕土",

            "寅": "少陽相火", "申": "少陽相火",

            "卯": "陽明燥金", "酉": "陽明燥金",

            "辰": "太陽寒水", "戌": "太陽寒水",

            "巳": "厥陰風木", "亥": "厥陰風木"

        };


        // 在泉對應表 (地支)

        const zaiQuanMap = {

            "子": "陽明燥金", "午": "陽明燥金",

            "丑": "太陽寒水", "未": "太陽寒水",

            "寅": "厥陰風木", "申": "厥陰風木",

            "卯": "少陰君火", "酉": "少陰君火",

            "辰": "太陰濕土", "戌": "太陰濕土",

            "巳": "少陽相火", "亥": "少陽相火"

        };


        // 大運對應表 (天干)

        const daYunMap = {

            "甲": "土運太過", "己": "土運不及",

            "乙": "金運不及", "庚": "金運太過",

            "丙": "水運太過", "辛": "水運不及",

            "丁": "木運不及", "壬": "木運太過",

            "戊": "火運太過", "癸": "火運不及"

        };


        function getFortuneData(year) {

            const baseYear = 1924; // 甲子

            const offset = (year - baseYear) % 60;

            const stemIdx = (year - 4) % 10;

            const branchIdx = (year - 4) % 12;

            

            const stem = stems[stemIdx];

            const branch = branches[branchIdx];

            const stemBranch = stem + branch;

            

            const siTian = siTianMap[branch];

            const zaiQuan = zaiQuanMap[branch];

            const daYun = daYunMap[stem];


            // 構建五運六氣描述

            const wuYun = `${stemBranch}年<br>${stem}為${daYun.slice(0,2)}（${daYun}）；<br>${siTian}司天；<br>${zaiQuan}在泉`;


            // 構建詳細特點 (模擬 V4 生成的邏輯)

            let features = [];

            features.push(`○ 該年為「${siTian}」司天，上半年氣候以${siTian.slice(2)}主事，氣候偏${siTian.slice(2,3) === '寒' ? '冷' : siTian.slice(2,3) === '燥' ? '乾' : '熱'}。`);

            

            if (daYun.includes("太過")) {

                features[0] += ` 受「${daYun.slice(0,2)}」之影響，氣候變化較為劇烈。`;

            } else {

                features[0] += ` 由於「${daYun.slice(0,2)}」不足，易受司天之氣克制。`;

            }


            features.push(`○ 下半年由「${zaiQuan}」在泉主導，${zaiQuan.slice(2)}之氣流行。`);

            

            let summary = "";

            if (daYun.includes("水")) summary = "全年水氣偏盛，寒氣較重，需注意冷熱交替。";

            else if (daYun.includes("火")) summary = "全年火氣偏旺，炎熱明顯，燥氣可能隨之。";

            else if (daYun.includes("木")) summary = "全年風氣大行，氣候多變且不穩定。";

            else if (daYun.includes("金")) summary = "全年燥氣偏盛，氣候趨於乾燥清冷。";

            else summary = "全年濕氣較重，氣候沈重潮濕，雨水可能較多。";


            features.push(`○ 總體而言，${summary}`);


            return {

                wuYun: wuYun,

                features: features

            };

        }


        const yearInput = document.getElementById('yearInput');

        const wuYunResult = document.getElementById('wuYunResult');

        const featureResult = document.getElementById('featureResult');


        function updateDisplay() {

            const year = parseInt(yearInput.value);

            if (year < 1927 || year > 2026 || isNaN(year)) {

                wuYunResult.innerHTML = "超出查詢範圍";

                featureResult.innerHTML = "請輸入 1927 至 2026 之間的年份。";

                return;

            }


            const data = getFortuneData(year);

            wuYunResult.innerHTML = data.wuYun;

            

            featureResult.innerHTML = data.features.map(f => `<p>${f}</p>`).join('');

        }


        yearInput.addEventListener('input', updateDisplay);


        // 初始執行

        window.onload = () => {

            const currentYear = new Date().getFullYear();

            // 系統指令要求默認顯示即時年份 2026

            yearInput.value = 2026; 

            updateDisplay();

        };

    </script>
